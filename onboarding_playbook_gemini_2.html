PostgreSQL schema for storing milestones in a real CRM, with a health-score view for CS tracking.

-- onboarding_milestones: tracks each milestone per customer CREATE TABLE onboarding_milestones ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), customer_id UUID NOT NULL REFERENCES customers(id), playbook_id UUID REFERENCES playbooks(id), phase INT NOT NULL CHECK (phase BETWEEN 1 AND 4), phase_name VARCHAR(100) NOT NULL, milestone_text TEXT NOT NULL, owner VARCHAR(50), -- 'CS', 'Client', 'Tech' status VARCHAR(20) DEFAULT 'pending' CHECK (status IN ('pending', 'in_progress', 'completed', 'skipped')), due_day INT, -- e.g. 7, 30, 60, 90 completed_at TIMESTAMPTZ, notes TEXT, created_at TIMESTAMPTZ DEFAULT now(), updated_at TIMESTAMPTZ DEFAULT now() ); -- Index for fast lookup per customer CREATE INDEX idx_milestones_customer ON onboarding_milestones (customer_id, phase, status); -- View: 90-day completion health score per customer CREATE VIEW v_onboarding_health AS SELECT customer_id, COUNT(*) AS total_milestones, SUM(CASE WHEN status = 'completed' THEN 1 ELSE 0 END) AS completed, ROUND( 100.0 * SUM(CASE WHEN status = 'completed' THEN 1 ELSE 0 END) / NULLIF(COUNT(*), 0), 1 ) AS completion_pct FROM onboarding_milestones GROUP BY customer_id;
